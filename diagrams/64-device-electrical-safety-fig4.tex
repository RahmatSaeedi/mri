\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small,
    ev/.style={draw, rounded corners=2pt, align=center, inner sep=4pt, minimum height=9mm},
    top/.style={ev, fill=red!14},
    inter/.style={ev, fill=orange!14},
    basic/.style={ev, fill=blue!8},
    gate/.style={draw, fill=gray!15, font=\footnotesize\bfseries, inner sep=2pt}]

  % Top event
  \node[top, text width=44mm] (top) {رویداد بالا: عبور جریان خطرناک از بدن};
  \node[gate, below=4mm of top] (gAND) {AND};
  \draw[thick] (top) -- (gAND);

  % Two children of AND
  \node[inter, below=12mm of gAND, xshift=-32mm, text width=36mm] (volt) {ولتاژ خطرناک به سطحی قابل‌لمس می‌رسد};
  \node[inter, below=12mm of gAND, xshift=32mm, text width=36mm] (path) {تماس شخص و وجود مسیر به زمین (محیطی)};
  \draw[thick] (gAND.south) -- (volt.north);
  \draw[thick] (gAND.south) -- (path.north);

  % OR gate under "voltage reaches surface"
  \node[gate, below=4mm of volt] (gOR) {OR};
  \draw[thick] (volt) -- (gOR);
  \node[basic, below=11mm of gOR, xshift=-22mm, text width=30mm] (mop1) {خرابی عایق پایه (MOP1) -- $p_1$};
  \node[basic, below=11mm of gOR, xshift=22mm, text width=30mm] (mop2) {خرابی عایق دوم / PE (MOP2) -- $p_2$};
  \draw[thick] (gOR.south) -- (mop1.north);
  \draw[thick] (gOR.south) -- (mop2.north);

  % Controls box
  \node[draw, rounded corners=3pt, fill=green!6, align=left, below=14mm of path, xshift=4mm, text width=58mm] (ctrl)
    {\textbf{کنترل‌ها (هر یک یک شاخه را قطع می‌کند):}\\
     $\bullet$ دو MOP مستقل $\Rightarrow$ نیاز به خرابی هم‌زمان\\
     $\bullet$ اتصال PE $\Rightarrow$ ولتاژ لمسی را پایین نگه می‌دارد\\
     $\bullet$ RCD 30\,mA $\Rightarrow$ جریان $\times$ زمان را محدود می‌کند\\
     $\bullet$ LOTO هنگام سرویس $\Rightarrow$ انرژی صفر پیش از تماس};

\end{tikzpicture}}
