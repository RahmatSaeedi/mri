  % macro to draw a small frame with z' (up) and y' (right); x' = B_eff out of page
  \tikzset{
    miniframe/.pic={
      \draw[->, thick] (0,-1.4) -- (0,1.7) node[above] {$z'$};
      \draw[->, thick] (-1.4,0) -- (1.7,0) node[right] {$y'$};
    }
  }

  % ---------- Panel A: initial M = M0 along z' ----------
  \begin{scope}[shift={(0,0)}]
    \pic {miniframe};
    \draw[->, very thick, blue!70!black] (0,0) -- (0,1.4) node[right] {$\mathbf{M}=M_0$};
    \node[font=\footnotesize] at (0.15,-2.1) {حالت اولیه};
  \end{scope}

  % ---------- Panel B: after 90 deg pulse -> transverse ----------
  \begin{scope}[shift={(4.6,0)}]
    \pic {miniframe};
    \draw[->, very thick, blue!70!black] (0,0) -- (1.4,0) node[above right] {$\mathbf{M}$};
    % rotation arc from z' to y'
    \draw[->, red!70!black] (0,1.0) arc (90:8:1.0);
    \node[red!70!black, font=\footnotesize] at (1.0,1.0) {$90^\circ$};
    \node[font=\footnotesize, align=center] at (0.15,-2.1) {پالس $90^\circ$\\(عرضی)};
  \end{scope}

  % ---------- Panel C: after 180 deg pulse -> inverted ----------
  \begin{scope}[shift={(9.2,0)}]
    \pic {miniframe};
    \draw[->, very thick, blue!70!black] (0,0) -- (0,-1.4) node[right] {$\mathbf{M}$};
    % rotation arc from +z' down to -z'
    \draw[->, red!70!black] (0.95,0.32) arc (18:-198:1.0);
    \node[red!70!black, font=\footnotesize] at (1.25,0.9) {$180^\circ$};
    \node[font=\footnotesize, align=center] at (0.15,-2.1) {پالس $180^\circ$\\\RL{(وارون‌شده، $-z'$)}};
  \end{scope}

  % shared note: rotation about x' (= B_eff, out of page)
  \node[font=\footnotesize] at (4.6,2.55)
