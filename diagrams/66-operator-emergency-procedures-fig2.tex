\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small,
    contact/.style={draw, rounded corners=2pt, align=center, inner sep=4pt,
                    fill=blue!8, minimum width=1.9cm, minimum height=1.0cm},
    relay/.style={draw, rounded corners=2pt, align=center, inner sep=4pt,
                  fill=green!14, minimum width=2.3cm, minimum height=1.0cm},
    src/.style={draw, circle, align=center, inner sep=1pt, fill=orange!18,
                minimum size=1.1cm},
    nc/.style={font=\footnotesize, text=blue!45!black},
    wire/.style={thick},
  ]

  % supply
  \node[src] (vcc) {$+24$\,V};

  % series chain of normally-closed contacts
  \node[contact, right=0.9cm of vcc] (estop) {توقف\\اضطراری};
  \node[contact, right=0.7cm of estop] (door) {درِ\\کابینت};
  \node[contact, right=0.7cm of door] (temp) {دمای\\مجاز};
  \node[contact, right=0.7cm of temp] (ball) {گویِ فشاریِ\\بیمار};
  \node[contact, right=0.7cm of ball] (wd) {سگِ نگهبان};
  \node[relay, right=0.9cm of wd] (relay) {رلهٔ مجوزِ\\توان};

  % NC labels under contacts
  \foreach \n in {estop,door,temp,ball,wd}{
    \node[nc, below=1pt of \n] {(NC)};
  }

  % series wiring
  \draw[wire] (vcc) -- (estop);
  \draw[wire] (estop) -- (door);
  \draw[wire] (door) -- (temp);
  \draw[wire] (temp) -- (ball);
  \draw[wire] (ball) -- (wd);
  \draw[wire] (wd) -- (relay);

  % output of relay -> enables gradient/RF power
  \draw[->, very thick] (relay.east) -- ++(1.3,0)
        node[right, align=left] {مجوزِ\\توان};

  % explanatory note
  \node[align=center, font=\footnotesize, below=1.7cm of temp, text width=13cm,
        draw, rounded corners=2pt, fill=red!7, inner sep=6pt] (note)
    {\textbf{ایمن در خرابی:} هر کنتاکتِ باز (فشردنِ دکمه، بازبودنِ در، فراگرمایی،
     فراخوانِ بیمار، وقفهٔ سگِ نگهبان) حلقهٔ سری را می‌گسلد\\
     $\Rightarrow$ رله رها می‌شود $\Rightarrow$ خروجیِ گرادیان / RF به \textbf{صفر} می‌رسد.
     افتِ $+24$\,V یا سیمِ گسسته نیز رله را رها می‌کند.};

\end{tikzpicture}}
