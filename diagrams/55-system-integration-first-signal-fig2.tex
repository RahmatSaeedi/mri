    pt/.style={circle, draw, fill=yellow!85!orange, inner sep=0pt, minimum size=4.4mm, font=\footnotesize\bfseries},

  % --- Forward (TX) chain ---

  \node[blk, right=7mm of rfpa] (tr1) {T/R};
  \node[blk, right=7mm of tr1] (coil1) {سیم‌پیچ};
  \node[right=9mm of coil1] (samp1) {\textbf{نمونه}};

  \draw[->] (nco) -- (dac);
  \draw[->] (dac) -- (rfpa);
  \draw[->] (rfpa) -- (tr1);
  \draw[->] (tr1) -- (coil1);
  \draw[->] (coil1) -- (samp1);

  % --- Reverse (RX) chain ---
  \node[rxblk, below=18mm of nco] (fft) {FFT /\\بازسازی};

  \node[rxblk, right=7mm of lna] (tr2) {T/R};
  \node[rxblk, right=7mm of tr2] (coil2) {سیم‌پیچ};

  \draw[<-] (fft) -- (adc);
  \draw[<-] (adc) -- (ddc);
  \draw[<-] (ddc) -- (lna);
  \draw[<-] (lna) -- (tr2);
  \draw[<-] (tr2) -- (coil2);

  % MR signal coupling from sample down into the RX coil
  \draw[->, thick] (samp1.south) -- ++(0,-9mm) -| (coil2.north)
        node[pos=0.25, right, font=\scriptsize\itshape, text=black!70]{سیگنال MR};

  % --- Injection / monitoring points ---
  \node[pt] (p1) at ($(dac.south)+(0,-5mm)$) {1};
  \draw[densely dotted] (dac.south) -- (p1);
  \node[pt] (p2) at ($(rfpa.north)+(0,5mm)$) {2};
  \draw[densely dotted] (rfpa.north) -- (p2);
