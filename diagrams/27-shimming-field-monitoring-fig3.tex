% NMR field camera (conceptual): an array of small NMR probes distributed on
% a sphere around the DSV. Each probe reports the local Larmor frequency via
% its FID phase; solving the linear system gives the true k-space trajectory
% plus higher-order harmonic dynamics for reconstruction correction.
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\tikzset{every node/.append style={execute at begin node=\setRTL}}
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]
  % --- Left: spherical arrangement of probes around the DSV ---
  \def\R{2.4}
  % Bounding sphere (dashed)
  \draw[dashed, gray!55] (0,0) circle (\R);
  % DSV at center
  \filldraw[red!12, draw=red!60!black, thick] (0,0) circle (0.7);
  \node[red!55!black, align=center] at (0,0) {\textbf{DSV}};

  % Probes around the sphere (16 shown), as filled dots
  \foreach \i in {0,...,15} {
    \pgfmathsetmacro{\ang}{\i*22.5 + 11.25}
    \filldraw[blue!70!black] (\ang:\R) circle (2.1pt);
  }
  % Label a couple of probes
  \node[blue!60!black, font=\footnotesize] at (101.25:\R+0.45) {پروبِ $j$};
  \node[blue!60!black, font=\footnotesize] at (-11.25:\R+0.55) {$\dots$ ۱۶ پروب};

  % Position vector to one probe
  \draw[->, thick, blue!60!black] (0,0) -- (56.25:\R)
    node[midway, above left, font=\footnotesize] {$\mathbf{r}_j$};

  \node[align=center, font=\footnotesize, below=2mm] at (0,-\R-0.2)
    {پروب‌های NMR روی یک کره};

  % --- Right: a single-probe detail box ---
  \node[draw=blue!55!black, thick, rounded corners, fill=blue!5,
        text width=40mm, align=left, inner sep=5pt] (det) at (5.8,1.1)
    {\textbf{هر پروب:}\\[2pt]
     $\bullet$ قطرهٔ NMR \LR{($\mu$L، $^{19}$F\,/\,$^{1}$H)}\\
     $\bullet$ سیم‌پیچ گیرندهٔ کوچک\\
     $\bullet$ پیش‌تقویت‌کننده};

  % Connect detail box to a probe
  \draw[->, blue!55!black, thick] (33.75:\R) .. controls (3.4,0.9) .. (det.west);

  % --- Bottom: signal-processing chain ---
  \node[draw=black!45, rounded corners, fill=green!8, align=center,
        text width=34mm, inner sep=4pt] (fid) at (5.8,-1.4)
    {FIDِ هر پروب\\$\Rightarrow$ فاز $\varphi_j(t)$};
  \node[draw=black!45, rounded corners, fill=orange!10, align=center,
        text width=46mm, inner sep=4pt, below=5mm of fid] (solve)
    {حلّ $\mathbf{k}(t)=\mathbf{H}_{\mathrm{probe}}^{+}\,\varphi(t)$\\
     $\Rightarrow$ مسیر واقعیِ k-space $+$\\دینامیکِ هارمونیکِ مرتبه‌بالا};

  \draw[->, thick] (det.south) -- (fid.north);
  \draw[->, thick] (fid.south) -- (solve.north);

  % phase-integral reminder
  \node[font=\footnotesize, gray!60!black, align=center, below=4mm of solve, text width=46mm]
    {$\varphi_j(t)=\gamma\!\int_0^t\! B_z(\mathbf{r}_j,t')\,dt'$};
\end{tikzpicture}}
\end{document}
