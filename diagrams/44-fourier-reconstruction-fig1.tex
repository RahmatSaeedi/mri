% Standalone TikZ figure: gridding pipeline (forward gridding -> image).
% Five-stage flowchart for reconstructing non-Cartesian (radial/spiral)
% k-space: scattered samples -> density compensation -> KB convolution onto
% an oversampled grid -> IFFT -> crop to FOV + roll-off correction -> image.
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\babelsublr{\begin{tikzpicture}[
    >=Stealth, font=\small,
    box/.style={rectangle, rounded corners=2pt, draw=black, very thick,
                align=center, minimum height=12mm, minimum width=30mm,
                fill=blue!6, inner sep=4pt},
    op/.style={ellipse, draw=black, very thick, align=center,
               fill=orange!12, inner sep=2pt, minimum height=10mm},
    flow/.style={-{Stealth[length=3mm]}, very thick}]

  % ---------- top row: input -> DCF -> convolution ----------
  \node[box] (data) {نمونه‌های پراکنده\\ $s(\mathbf{k}_j)$\\ شعاعی / مارپیچی};
  \node[op, right=14mm of data] (dcf) {جبران چگالی\\ $s(\mathbf{k}_j)\cdot w_j$};
  \node[box, right=14mm of dcf, fill=green!8] (conv)
        {پیچش روی\\ شبکهٔ بیش‌نمونه‌برداری‌شده\\ هستهٔ $C(\mathbf{k})$ \RL{($\alpha N$)}};

  \draw[flow] (data) -- (dcf);
  \draw[flow] (dcf) -- (conv);

  % ---------- down to IFFT ----------
  \node[box, below=12mm of conv, fill=green!8] (ifft)
        {IFFT روی\\ شبکهٔ بیش‌نمونه‌برداری‌شده};
  \draw[flow] (conv) -- (ifft);

  % ---------- IFFT -> crop+rolloff -> image (right to left) ----------
  \node[box, below=12mm of dcf] (crop)
        {برش به FOV\\ $+$ تصحیح افت\\ تقسیم بر $c(\mathbf{r})$};
  \node[box, below=12mm of data, fill=blue!12] (image)
        {تصویر\\ $m(\mathbf{r})$};

  \draw[flow] (ifft) -- (crop);
  \draw[flow] (crop) -- (image);

  % ---------- title ----------
  \node[font=\bfseries, above=4mm of dcf]
        {خط‌لولهٔ شبکه‌بندی پیشرو: فضای‑کی غیردکارتی $\rightarrow$ تصویر};

\end{tikzpicture}}
\end{document}
