% Standalone TikZ figure: GRAPPA kernel geometry for R=2.
% Two channels (c' source, c target). Acquired source lines (filled), missing
% lines (open), and the target point reconstructed as a weighted combination
% of neighboring acquired points across all channels.
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\tikzset{every node/.append style={execute at begin node=\setRTL}}
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]

  \def\dx{0.85}    % horizontal spacing along k_y
  % column k_y positions: 0..6 -> alternating acquired / missing
  % acquired at even indices (0,2,4,6); missing at odd (1,3,5)
  \def\yC{2.6}     % row for channel c'
  \def\yc{0.0}     % row for channel c

  % --- k_y axis arrow ---
  \draw[->] (-0.5,3.9) -- (6.2*\dx,3.9) node[right, font=\scriptsize] {$k_y$ (کدگذاری فازی)};

  % --- channel c' (source row, top) ---
  \node[left, font=\scriptsize] at (-0.4,\yC) {کانال $c'$:};
  \foreach \i in {0,2,4,6} { \fill (\i*\dx,\yC) circle (0.10); }            % acquired
  \foreach \i in {1,3,5} { \draw (\i*\dx,\yC) circle (0.10); }              % missing

  % --- channel c (target row, bottom) ---
  \node[left, font=\scriptsize] at (-0.4,\yc) {کانال $c$:};
  \foreach \i in {0,4,6} { \fill (\i*\dx,\yc) circle (0.10); }              % acquired
  \draw[fill=red!75!black] (2*\dx,\yc) circle (0.12);                        % target (reconstructed)
  \foreach \i in {1,3,5} { \draw (\i*\dx,\yc) circle (0.10); }              % missing

  % --- contributing source points (neighbors of the target, both channels) ---
  % from channel c' : acquired neighbors at i=0,2,4 ; from channel c : i=0,4
  \foreach \i in {0,2,4} {
    \draw[->, blue!55!black, thin] (\i*\dx,\yC) -- (2*\dx,\yc);
  }
  \foreach \i in {0,4} {
    \draw[->, blue!55!black, thin] (\i*\dx,\yc) -- (2*\dx,\yc);
  }

  % --- legend ---
  \begin{scope}[shift={(0.1,-1.5)}]
    \fill (0,0) circle (0.10);
    \node[right, font=\scriptsize] at (0.2,0) {گرفته‌شده (منبع)};
    \draw (3.0,0) circle (0.10);
    \node[right, font=\scriptsize] at (3.2,0) {خطِ ازدست‌رفته};
    \draw[fill=red!75!black] (5.4,0) circle (0.12);
    \node[right, font=\scriptsize] at (5.6,0) {هدف در $c$};
  \end{scope}

  % --- explanatory note ---
  \node[font=\scriptsize, align=center, text width=8cm] at (2.6*\dx,-2.45)
    {نقطهٔ ازدست‌رفته $=\sum_{c'}\sum_b w^{(r)}_{c,c',b}\,d_{c'}$
     (ترکیبِ وزن‌دارِ همسایه‌ها از همهٔ کانال‌ها)};

\end{tikzpicture}}
\end{document}
