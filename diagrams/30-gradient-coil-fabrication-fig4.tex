    note/.style={font=\scriptsize, gray, align=center}]

  \node[box] (cur) {جریانِ گرادیان $I(t)$};
  \node[box, right=3.0cm of cur, fill=red!10] (force) {نیروی لورنتس $F(t)$};
  \node[box, below=1.6cm of force, fill=orange!12] (vib) {تغییرشکل / ارتعاشِ\\سازهٔ سیم‌پیچ};
  \node[box, below=1.6cm of vib, fill=green!12] (rad) {حرکتِ سطح\\$\rightarrow$ تابشِ صوتی};
  \node[box, below=1.6cm of rad, fill=gray!12] (spl) {فشارِ صوت\\SPL در گوشِ بیمار};

  % current -> force
  \draw[->, very thick] (cur) -- node[above, font=\scriptsize]{$\times\,B_0$} (force);
  \node[note, below=1pt of cur] {$dI/dt$ تند};

  % force -> vibration
  \draw[->, very thick] (force) -- (vib);
  \node[note, right=2pt of $(force)!0.5!(vib)$] {نیروی\\پالسیِ نوسانی};

  % vibration -> radiation
  \draw[->, very thick] (vib) -- (rad);
  \node[note, right=2pt of $(vib)!0.5!(rad)$] {مُدهای\\تشدیدِ مکانیکی};

  % radiation -> SPL
  \draw[->, very thick] (rad) -- (spl);

  % feedback emphasis arc from current directly into vibration
  \draw[->, dashed, thick, blue!50!black]
     (cur.south) .. controls +(0,-2.0) and +(-2.4,0) .. (vib.west);

\end{tikzpicture}}
