% Standalone TikZ figure: SSDU self-supervised learning. The measured k-space
% set Omega is split into Theta (input to the unrolled network) and Lambda
% (held-out target). The network output m-hat is re-encoded to predict the
% Lambda points; the loss is computed only on Lambda (points the network did
% not see), analogous to cross-validation.
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\tikzset{every node/.append style={execute at begin node=\setRTL}}
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small,
    io/.style={draw, fill=gray!12, rounded corners=2pt, thick, align=center},
    net/.style={draw, rounded corners=2pt, thick, align=center, fill=blue!6,
                minimum height=12mm},
    op/.style={draw, rounded corners=2pt, thick, align=center, fill=orange!12}]

  % Source set Omega
  \node[io] (omega) {فضای‑کیِ اندازه‌گیری‌شده\\ $\Omega=\Theta\cup\Lambda$};

  % Split nodes
  \node[io, right=18mm of omega, yshift=11mm, fill=blue!10] (theta)
        {$\Theta$ (ورودی)};
  \node[io, right=18mm of omega, yshift=-11mm, fill=green!12] (lambda)
        {$\Lambda$ (هدف)};

  \draw[->, thick] (omega.east) |- (theta.west);
  \draw[->, thick] (omega.east) |- (lambda.west);
  \node[font=\footnotesize, above=0pt of $(omega)!0.5!(theta)$] {تقسیم تصادفی};

  % Unrolled network on the Theta branch
  \node[net, right=14mm of theta] (net) {شبکهٔ\\ بازشده};
  \node[io, right=12mm of net] (mhat) {$\hat{\mathbf{m}}$};
  \node[op, right=12mm of mhat] (enc) {$E_\Lambda\,\hat{\mathbf{m}}$\\\footnotesize پیش‌بینی در $\Lambda$};

  \draw[->, thick] (theta) -- (net);
  \draw[->, thick] (net) -- (mhat);
  \draw[->, thick] (mhat) -- (enc);

  % Loss node comparing prediction and held-out measurements
  \node[op, fill=red!10, below=12mm of enc] (loss)
        {اتلاف فقط روی $\Lambda$\\\footnotesize(نقاطی که شبکه ندیده)};

  \draw[->, thick] (enc) -- (loss);
  \draw[->, thick] (lambda.south) |- (loss.west);

  % annotation
  \node[below=3mm of loss, align=center, font=\footnotesize\itshape, text width=70mm]
        {سازگاری با داده تنها از $\Theta$ استفاده می‌کند؛ نظارت از
         $\Lambda$ِ کنارگذاشته می‌آید -- مانند اعتبارسنجی متقابل};

\end{tikzpicture}}
\end{document}
