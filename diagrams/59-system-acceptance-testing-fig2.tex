  % helper: draw a 4x2 grid of filled holes at column spacing s, dot radius rad
  % parameters: #1 = x-origin, #2 = spacing, #3 = dot radius, #4 = label
  \newcommand{\holegroup}[4]{%
    \foreach \i in {0,1,2,3} {%
      \foreach \j in {0,1} {%
        \fill[blue!60!black] ($(#1,0)+(\i*#2, \j*#2)$) circle (#3);
      }%
    }%
    \node[font=\footnotesize] at ($(#1,0)+(1.5*#2, -0.7)$) {#4};
  }

  % three groups, decreasing spacing -> harder to resolve
  \holegroup{0}{0.55}{0.13}{$1.1$\,mm}
  \holegroup{3.0}{0.45}{0.11}{$1.0$\,mm}
  \holegroup{5.6}{0.34}{0.085}{$0.9$\,mm}

  % "finer = harder to separate" arrow
  \draw[->, thick, red!60!black] (0.0,1.45) -- (6.9,1.45)
        node[midway, above, font=\footnotesize, text=red!60!black]
        {فاصلهٔ ریزتر $\Rightarrow$ تفکیک دشوارتر};

  % acceptance note
  \node[align=center, font=\footnotesize] at (3.45,-1.5)
    {پذیرش ACR: گروه $1.0$\,mm باید کاملاً تفکیک‌پذیر باشد.};

  % title
  \node[font=\small\bfseries] at (3.45,2.1)
    {الگوی سوراخ‌های تفکیک‌پذیری (نمای بزرگ‌نمایی‌شده)};

\end{tikzpicture}}
