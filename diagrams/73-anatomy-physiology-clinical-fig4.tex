\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]

  % ---- ECG trace with QRS / R peaks ----
  % baseline at y=0; R-waves at x = 0, 3, 6, 9
  \def\ymax{1.6}
  \draw[->, thick] (-0.4,0) -- (11.3,0) node[right] {زمان};
  \node[left] at (-0.4,0.7) {ECG};

  \foreach \x in {0,3,6,9}{
    % P bump, QRS spike, T bump
    \draw[very thick, red!70!black]
      (\x-0.9,0) .. controls (\x-0.75,0.18) and (\x-0.6,0.18) .. (\x-0.45,0)  % P
      -- (\x-0.18,0) -- (\x-0.08,-0.22) -- (\x,\ymax) -- (\x+0.08,-0.25) -- (\x+0.18,0) % QRS
      .. controls (\x+0.45,0.28) and (\x+0.7,0.28) .. (\x+0.95,0); % T
    \node[red!70!black, font=\footnotesize\bfseries] at (\x,\ymax+0.28) {R};
  }

  % RR interval bracket between first two R-waves
  \draw[<->] (0,-0.7) -- (3,-0.7) node[midway, fill=white, font=\footnotesize] {یک چرخهٔ قلبی (RR)};

  % ---- k-space / cardiac phase filling under the trace ----
  \def\ka{-2.3}
  \node[left, align=right] at (-0.4,\ka+0.45) {اکتساب\\(فازهای\\قلبی)};
  % within each RR, a few colored segments = cardiac phases sampled
  \definecolor{ph1}{RGB}{31,119,180}
  \definecolor{ph2}{RGB}{44,160,44}
  \definecolor{ph3}{RGB}{255,127,14}
  \definecolor{ph4}{RGB}{148,103,189}
  \foreach \x in {0,3,6,9}{
    \fill[ph1] (\x+0.05,\ka) rectangle (\x+0.75,\ka+0.9);
    \fill[ph2] (\x+0.78,\ka) rectangle (\x+1.48,\ka+0.9);
    \fill[ph3] (\x+1.51,\ka) rectangle (\x+2.21,\ka+0.9);
    \fill[ph4] (\x+2.24,\ka) rectangle (\x+2.94,\ka+0.9);
    \draw[gray, densely dotted] (\x,\ka) -- (\x,0);
  }
  % phase legend
  \node[font=\footnotesize, anchor=west] at (0.05,\ka-0.5)
    {هر رنگ = یک فاز قلبی؛ خطوط در طول چرخه‌ها انباشته می‌شوند تا یک حلقهٔ CINE کامل شود};

\end{tikzpicture}}
