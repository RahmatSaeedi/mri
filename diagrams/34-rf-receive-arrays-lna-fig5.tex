\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small,
    box/.style={draw, rounded corners=2pt, align=center, fill=blue!4}]

  % --- Resonant loop with distributed capacitors ---
  \node[draw, rounded corners=3pt, minimum width=92mm, minimum height=14mm, fill=blue!5] (loop) at (0,0) {};
  \node[anchor=north, font=\footnotesize] at ([yshift=-0.5mm]loop.north)
       {حلقهٔ تشدید با خازن‌های توزیع‌شده};
  % capacitor symbols along the loop
  \foreach \x in {-3.4,-1.8,-0.2,1.4} {
    \draw[thick] (\x,-0.15) -- (\x,0.15);
    \draw[thick] (\x+0.22,-0.15) -- (\x+0.22,0.15);
  }
  \node[font=\footnotesize] at (3.3,-0.05) {$L_{\text{loop}},\,R$};
  \node[font=\footnotesize] at (-0.9,-0.5) {$C_1 \dots C_4$ (توزیع‌شده)};

  % --- two branches down ---
  \coordinate (ld) at (-2.6,-0.7);   % left tap (active detune)
  \coordinate (rd) at (2.6,-0.7);    % right tap (match network)
  \draw[thick] (loop.south -| ld) -- (ld);
  \draw[thick] (loop.south -| rd) -- (rd);

  % --- Active detune block (left) ---
  \node[box, minimum width=24mm, minimum height=11mm, below=6mm of ld, fill=red!5] (det)
       {ناکوک‌سازیِ فعال\\$L_{\text{det}}$، PIN};
  \draw[->, thick] (ld) -- (det.north);
  \node[right=1mm of det, font=\footnotesize, align=left] {$+\,I_{\text{bias}}$};
  % passive protection note
  \node[box, minimum width=24mm, minimum height=9mm, below=5mm of det, fill=red!3] (prot)
       {حفاظتِ غیرفعال\\(دیودهای پادموازی)};
  \draw[thick] (det.south) -- (prot.north);

  % --- Match / decoupling block (right) ---
  \node[box, minimum width=26mm, minimum height=11mm, below=6mm of rd, fill=orange!8] (match)
       {تطبیق $+$ شبکهٔ مسدودسازی\\$C_m,\,L_m$};
  \draw[->, thick] (rd) -- (match.north);

  % --- LNA ---
  \node[box, minimum width=34mm, minimum height=11mm, below=8mm of match, fill=green!7] (lna) {LNA};
  \node[right=1.5mm of lna, font=\footnotesize, align=left]
       {NF $\approx 0.5$ dB، $G\approx 28$ dB،\\$Z_{\text{in}}\to 0$};
  \draw[->, thick] (match.south) -- (lna.north);

  % --- Cable trap / balun ---
  \node[box, minimum width=34mm, minimum height=9mm, below=8mm of lna, fill=black!4] (trap)
       {تلهٔ کابل (balun)};
  \draw[->, thick] (lna.south) -- (trap.north);

  % --- Coax to digital receiver ---
  \draw[->, thick] (trap.south) -- ++(0,-0.9)
       node[below, font=\footnotesize] {coax $\to$ گیرندهٔ دیجیتال};

\end{tikzpicture}}
