\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}

\usepackage{pgfplots}

\begin{document}
\setRTL
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]
  % --- copper matrix (round wire cross-section) ---
  \def\R{2.6}
  \fill[orange!30!yellow!60] (0,0) circle (\R);
  \draw[thick, orange!50!black] (0,0) circle (\R);

  % --- NbTi filaments arranged on a hex-ish grid inside the circle ---
  \foreach \x in {-1.8,-1.2,-0.6,0,0.6,1.2,1.8} {
    \foreach \y in {-1.8,-1.2,-0.6,0,0.6,1.2,1.8} {
      % offset alternate rows for a packed look
      \pgfmathsetmacro{\xo}{\x + ( (mod(round((\y)/0.6),2)==0) ? 0 : 0.3 )}
      \pgfmathsetmacro{\rr}{sqrt(\xo*\xo + \y*\y)}
      \ifdim \rr pt < 2.05pt
        \fill[blue!70!black] (\xo,\y) circle (0.18);
      \fi
    }
  }

  % --- callout: a single filament ---
  \draw[->, thick] (3.9,1.4) -- (0.9,0.85);
  \node[anchor=west, align=left] at (3.9,1.55)
    {رشتهٔ NbTi\\ \LR{$(\sim 5\mbox{--}50\,\mu\mathrm{m})$}};

  % --- callout: inter-filament copper ---
  \draw[->, thick] (3.9,-0.2) -- (1.45,0.05);
  \node[anchor=west, align=left] at (3.9,-0.2)
    {Cu میان‌رشته‌ای};

  % --- callout: copper matrix (stabilizer) ---
  \draw[->, thick] (-3.9,1.9) -- (-1.95,1.5);
  \node[anchor=east, align=right] at (-3.9,1.9)
    {بستر Cu\\(پایدارساز)};

  % --- bottom annotations ---
  \node[align=center, anchor=north] at (0,-2.95)
    {رشته‌ها: $10^2$ تا $10^4$ \quad|\quad نسبت Cu:SC $\approx 1{:}1$ تا $8{:}1$};

  % --- twist pitch indicator ---
  \draw[->, thick] (-3.9,-1.6) -- (-1.95,-1.1);
  \node[anchor=east, align=right] at (-3.9,-1.6)
    {تابیده با\\ گام $L_p$};
\end{tikzpicture}}
