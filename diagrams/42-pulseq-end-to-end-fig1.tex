  %================= LEFT: DESIGN PHASE (host, offline) =================
  \node[box] (idea)  at (0,7)   {ایدهٔ فیزیکی\\(مثلاً GRE)};
  \node[box] (script) at (0,5.4) {اسکریپت PyPulseq\\(\texttt{Opts}، ساخت بلوک‌ها)};

  \node[box] (sim)   at (0,2.2)  {شبیه‌ساز بلاخ (فصل~۱۹)\\پیش‌بینی $k$-space و تصویر};

  \draw[flow] (idea) -- (script);
  \draw[flow] (script) -- (check);
  \draw[flow] (check) -- node[right,font=\footnotesize]{(اختیاری)} (sim);

  \node[grp, fit=(idea)(script)(check)(sim)] (designgrp) {};
  \node[font=\bfseries, align=center] at (0,8) {فاز طراحی\\\footnotesize\mdseries (میزبان، برون‌خط)};

  %================= RIGHT: EXECUTION PHASE (console, real-time) =================
  \node[box] (interp) at (9,7)   {مفسّر Pulseq روی کنسول\\(Python/C++ --- فصل~۴۱)};
  \node[box] (evtab)  at (9,5.4) {جدول رویدادهای زمان‌بندی‌شده};
  \node[box] (fpga)   at (9,3.8) {بارگذاری در میان‌افزار FPGA\\(RF LUT، grad DAC، ADC)\\فصل~۳۸/۳۹};
  \node[box] (raw)    at (9,2.2) {دادهٔ خام ADC (I/Q)\\TX $\to$ sample $\to$ RX};
  \node[box] (recon)  at (9,0.6) {بازشبکه‌بندی به $k$-space و بازسازی\\ذخیره در ISMRMRD/HDF5 (فصل~۴۳)};

  \draw[flow] (interp) -- (evtab);
  \draw[flow] (evtab) -- (fpga);
  \draw[flow] (fpga) -- (raw);
  \draw[flow] (raw) -- (recon);

  \node[grp, fit=(interp)(evtab)(fpga)(raw)(recon)] (execgrp) {};
  \node[font=\bfseries, align=center] at (9,8) {فاز اجرا\\\footnotesize\mdseries (کنسول، بی‌درنگ)};

  %================= OPEN-FORMAT BOUNDARY =================
  \draw[flow, line width=1pt] (designgrp.east) -- node[above, font=\footnotesize, align=center]{فایل \texttt{.seq}}

  \node[font=\itshape\footnotesize, align=center] at (4.5,8) {مرز\\قالب‌باز};

  %================= FEEDBACK LOOP (compare / debug) =================
  \draw[flow, dashed, red!70!black]

    -| (designgrp.south);
