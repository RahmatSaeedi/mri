% an RF event, up to three gradient events (Gx, Gy, Gz) and one ADC event,
% with their defining parameters and the block-duration / overlap constraint.

\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}

\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small,
    event/.style={rectangle, rounded corners=2pt, draw, very thick,
                  minimum width=2.6cm, minimum height=0.85cm,
                  align=center, font=\small\bfseries},
    detail/.style={anchor=west, font=\scriptsize, gray!45!black, align=left}]

  % ---- title ----
  \node[font=\bfseries] (title) at (0,0.95) {یک بلوک Pulseq (یک بازهٔ زمانی واحد)};

  % ---- channel rows (top to bottom): RF, Gx, Gy, Gz, ADC ----
  \def\rowsep{1.25}
  \node[event, fill=blue!18,   draw=blue!70!black]   (rf)  at (0,0)            {RF};
  \node[event, fill=red!18,    draw=red!65!black]     (gx)  at (0,-\rowsep)     {$G_x$};
  \node[event, fill=red!12,    draw=red!55!black]     (gy)  at (0,-2*\rowsep)   {$G_y$};
  \node[event, fill=red!8,     draw=red!45!black]     (gz)  at (0,-3*\rowsep)   {$G_z$};
  \node[event, fill=purple!16, draw=purple!70!black]  (adc) at (0,-4*\rowsep)   {ADC};

  % ---- per-event parameter annotations ----
  \node[detail] at (rf.east)  [xshift=4pt] {پوش شکل‌موج $+$ بسامد/فاز حامل\\$+$ شیب $+$ تأخیر};
  \node[detail] at (gx.east)  [xshift=4pt] {ذوزنقه‌ای \emph{یا} دلخواه\\\RL{(دامنه، بالاروی، سطح ثابت$\dots$)}};
  \node[detail] at (gy.east)  [xshift=4pt] {ذوزنقه‌ای \emph{یا} دلخواه};
  \node[detail] at (gz.east)  [xshift=4pt] {ذوزنقه‌ای \emph{یا} دلخواه};
  \node[detail] at (adc.east) [xshift=4pt] {نمونه‌ها $+$ زمان توقف $+$ تأخیر $+$ فاز};

  % ---- gradient grouping brace ----
  \draw[decorate, decoration={brace, amplitude=5pt}, gray!55]
    (gx.west) ++(-0.25,0.18) -- ($(gz.west)+(-0.25,-0.18)$)
    node[midway, left=6pt, font=\scriptsize, gray!45!black, align=center, rotate=90, anchor=south]
    {تا ۳ محور گرادیان};

  % ---- enclosing block (fit box) ----
  \begin{scope}[on background layer]
    \node[draw, very thick, rounded corners=4pt, fill=gray!6,
          inner sep=12pt, fit=(rf)(gz)(adc)(title)] (block) {};
  \end{scope}

  % ---- constraint note below the block ----
  \node[below=8pt of block.south, anchor=north, align=center, font=\scriptsize\itshape,
        draw=gray!50, fill=yellow!12, rounded corners=2pt, inner sep=5pt]
    {قیدها: RF و ADC نباید هم‌پوشانی داشته باشند\\
     مدت بلوک $=$ طولانی‌ترین رویداد دربرگرفته};

\end{tikzpicture}}
\end{document}
