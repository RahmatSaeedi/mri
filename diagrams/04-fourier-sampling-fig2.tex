% Left: object rho(r) inside the field of view FOV, nominal resolution dx.
% Right: discretely sampled k-space s(k), step dk = 1/FOV, extent 2*k_max.

\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]

  % ---------- LEFT: image space rho(r) ----------
  \begin{scope}
    \node[align=center] at (1.75,4.2) {فضای تصویر $\rho(r)$};
    % FOV box
    \draw[thick] (0,0) rectangle (3.5,3.5);
    % object (smooth blob) inside the FOV
    \draw[fill=blue!12, draw=blue!70!black, very thick]
      plot[smooth cycle, tension=0.9]
      coordinates {(1.75,2.9) (2.6,2.3) (2.55,1.3) (1.75,0.8) (0.95,1.3) (0.9,2.3)};
    \node[blue!60!black] at (1.75,1.85) {جسم};
    % FOV double arrow
    \draw[<->] (0,-0.35) -- (3.5,-0.35) node[midway,below] {FOV};
    % resolution element dx
    \draw[<->] (0.05,3.78) -- (0.55,3.78);
    \node[above] at (0.3,3.78) {$\Delta x = \frac{1}{2k_{\max}}$};
  \end{scope}

  % ---------- ARROW (Fourier transform) ----------
  \draw[{Stealth[length=4mm]}-{Stealth[length=4mm]}, very thick]
        (4.2,1.75) -- node[above, align=center] {$\mathcal{F}$} (5.6,1.75);

  % ---------- RIGHT: k-space sampling grid ----------
  \begin{scope}[xshift=6.3cm]
    \node[align=center] at (1.75,4.2) {فضای‑کی $s(k)$};
    % k-space extent box
    \draw[thick] (0,0) rectangle (3.5,3.5);
    % sample grid points
    \foreach \i in {0,...,6}{
      \foreach \j in {0,...,6}{
        \fill[black] ({0.25+\i*0.5},{0.25+\j*0.5}) circle (1.4pt);
      }
    }
    % dk step (between two adjacent samples)
    \draw[<->, red!70!black, thick] (0.25,-0.35) -- (0.75,-0.35);
    \node[red!70!black, below] at (0.5,-0.4) {$\Delta k=\frac{1}{\mathrm{FOV}}$};
    % 2 k_max extent
    \draw[<->] (4.0,0.25) -- (4.0,3.25);
    \node[rotate=-90] at (4.4,1.75) {$2\,k_{\max}$};
  \end{scope}

  % ---------- bottom relations ----------
  \node[align=center] at (5.3,-1.4)
    {$k_{\max}\uparrow \Rightarrow \Delta x\downarrow$ (تفکیک‌پذیری بهتر)\\[2pt]
     $\Delta k\downarrow \Rightarrow \mathrm{FOV}\uparrow$ (میدان دید بزرگ‌تر)};

\end{tikzpicture}}
