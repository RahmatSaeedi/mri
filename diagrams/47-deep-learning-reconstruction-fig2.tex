% Standalone TikZ figure: unrolling a fixed number K of model-based iterations
% into a deep network. Each cascade = data-consistency (DC) step + learned CNN
% regularizer, trained end-to-end. Weights theta shared (MoDL) or independent
% (VarNet).
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small,
    cell/.style={draw, rounded corners=2pt, align=center, minimum height=15mm,
                 minimum width=27mm, thick, fill=blue!5},
    io/.style={draw, fill=gray!12, rounded corners=2pt, thick, align=center}]

  % Input chain
  \node[io] (d) {$\mathbf{d}$\\[-1pt]\footnotesize (فضای‑کی)};
  \node[io, right=10mm of d] (m0) {$\mathbf{m}^{0}=E^{H}\mathbf{d}$};
  \draw[->, thick] (d) -- (m0);

  % Iteration cells
  \node[cell, right=12mm of m0] (it1)
        {\textbf{تکرار ۱}\\[2pt]
         $\mathbf{z}^{1}=\mathrm{DC}(\mathbf{m}^{0},\mathbf{d})$\\
         $\mathbf{m}^{1}=\mathrm{CNN}_\theta(\mathbf{z}^{1})$};
  \node[cell, right=12mm of it1] (it2)
        {\textbf{تکرار ۲}\\[2pt]
         $\mathbf{z}^{2}=\mathrm{DC}(\mathbf{m}^{1},\mathbf{d})$\\
         $\mathbf{m}^{2}=\mathrm{CNN}_\theta(\mathbf{z}^{2})$};
  \node[right=10mm of it2] (dots) {$\cdots$};
  \node[cell, right=10mm of dots] (itK)
        {\textbf{تکرار $K$}\\[2pt]
         $\mathbf{z}^{K}=\mathrm{DC}(\cdots)$\\
         $\mathbf{m}^{K}=\mathrm{CNN}_\theta(\cdots)$};
  \node[io, right=12mm of itK] (out) {$\hat{\mathbf{m}}$};

  \draw[->, thick] (m0) -- (it1);
  \draw[->, thick] (it1) -- (it2);
  \draw[->, thick] (it2) -- (dots);
  \draw[->, thick] (dots) -- (itK);
  \draw[->, thick] (itK) -- (out);

  % Shared-weights brace beneath the cells
  \coordinate (lb) at ($(it1.south)+(0,-6mm)$);
  \coordinate (rb) at ($(itK.south)+(0,-6mm)$);
  \draw[decorate, decoration={brace, amplitude=4pt, mirror}, thick]
        ($(it1.south west)+(0,-3mm)$) -- ($(itK.south east)+(0,-3mm)$);
  \node[below=8mm of dots, align=center, font=\footnotesize]
        {وزن‌های $\theta$ مشترک (MoDL) یا مستقل در هر تکرار (VarNet)};
  \foreach \c in {it1,it2,itK}{
    \draw[densely dotted] (\c.south) -- ($(\c.south)+(0,-3mm)$);
  }

  % Legend
  \node[below=2mm of out, align=left, font=\footnotesize, anchor=north east,
        xshift=2mm] at ($(itK.south east)+(0,-14mm)$)
        {\textbf{DC} = گامِ سازگاری با داده (فیزیک: $E$، $E^{H}$)\\
         \textbf{CNN}$_\theta$ = منظم‌سازِ آموخته‌شده (جایگزینِ prox)};

\end{tikzpicture}}
\end{document}
