% Components of a post-market surveillance (PMS) system: field data sources feed
% event triage; serious events trigger mandatory reporting to the regulator,
% non-serious ones feed trend analysis; both converge on CAPA / recall and feed
% periodic reports (PSUR / PMCF / annual reports).
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\tikzset{every node/.append style={execute at begin node=\setRTL}}
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small,
  box/.style={draw, rounded corners=2pt, align=center, minimum height=11mm},
  src/.style={box, fill=blue!10, text width=92mm},
  dec/.style={diamond, draw, aspect=2.2, fill=yellow!16, align=center,
              inner sep=1pt, font=\footnotesize, text width=34mm},
  rep/.style={box, fill=red!12, text width=46mm, font=\footnotesize},
  trend/.style={box, fill=blue!8, text width=46mm, font=\footnotesize},
  capa/.style={box, fill=orange!16, very thick, text width=52mm, font=\footnotesize},
  per/.style={box, fill=green!12, text width=58mm, font=\footnotesize},
  lbl/.style={font=\scriptsize, fill=white, inner sep=1pt}]

  % Field data sources
  \node[src] (data) {\textbf{منابع دادهٔ میدانی}\\[1pt]
        {\footnotesize شکایت‌ها $\mid$ گزارش‌های خدمات $\mid$ تله‌متری $\mid$ ادبیات علمی $\mid$ بازخورد کاربر}};

  % Triage decision
  \node[dec, below=11mm of data] (tri) {رخداد جدیِ\\ گزارش‌پذیر؟};
  \draw[->, thick] (data) -- (tri);

  % Serious -> mandatory report
  \node[rep, below=18mm of tri, xshift=-32mm] (rep) {\textbf{گزارش الزامی به نهاد ناظر}\\[1pt]
        FDA: MDR (eMDR)\\ EU: مراقبت / MIR\\ IR-FDA: گزارش رخداد};
  \draw[->, thick] (tri.west) -| (rep.north) node[lbl, pos=0.25] {بله (جدی)};

  % Non-serious -> trend analysis
  \node[trend, below=18mm of tri, xshift=32mm] (trend) {\textbf{تحلیل روند}\\[1pt]
        به‌روزرسانی پروندهٔ ریسک};
  \draw[->, thick] (tri.east) -| (trend.north) node[lbl, pos=0.25] {خیر (نشانه / روند)};

  % Converge on CAPA
  \node[capa, below=16mm of tri] (capa) {\textbf{CAPA (فصل ۵۶)}\\[1pt]
        $+$ در صورت نیاز: فراخوان / FSCA};
  \draw[->, thick] (rep.south) |- (capa.west);
  \draw[->, thick] (trend.south) |- (capa.east);

  % Periodic reports
  \node[per, below=11mm of capa] (per) {\textbf{گزارش‌های دوره‌ای}\\[1pt]
        FDA: گزارش‌های سالانه\\ EU: PSUR $+$ PMCF};
  \draw[->, thick] (capa) -- (per);
\end{tikzpicture}}
\end{document}
