\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small, node distance=7mm]
  \tikzset{
    test/.style={rectangle, draw=blue!55!black, thick, fill=blue!6,
      text width=52mm, align=left, inner sep=4pt, rounded corners=2pt},
    crit/.style={rectangle, draw=green!50!black, thick, fill=green!9,
      text width=52mm, align=left, inner sep=4pt, rounded corners=2pt,
      font=\footnotesize},
    head/.style={rectangle, draw=black!55, very thick, fill=gray!12,
      text width=58mm, align=center, inner sep=5pt, rounded corners=3pt}
  }

  % Header
  \node[head] (h) {\textbf{رویهٔ آزمون و پذیرشِ شیم}\\[1pt]
    \footnotesize هدف: راستی‌آزماییِ یکنواختیِ $B_0$ پس از شیمِ غیرفعال\,+\,فعال
    و سلامتِ کانال‌ها.\\
    \footnotesize تجهیزات: فانتومِ کرویِ یکنواخت یا دوربین میدان؛
    نرم‌افزارِ تحلیلِ هارمونیک.};

  % Test 1
  \node[test, below=8mm of h] (t1) {\textbf{آزمون ۱ — یکنواختیِ پایه}
    (شیمِ فعال خاموش): نقشهٔ $B_0$ دواکویی روی DSV با قطر ۴۵\,cm؛ محاسبهٔ RMS و p2p.};
  \node[crit, below=2mm of t1] (c1) {قبول: RMS $<2$\,ppm،\ p2p $<10$\,ppm (بالینیِ ۱٫۵ تا ۳\,T).};

  % Test 2
  \node[test, below=6mm of c1] (t2) {\textbf{آزمون ۲ — بازده و
    خطی‌بودنِ کانال}: برای هر کانال، پلهٔ جریان، نقشه‌برداری، تحلیلِ هارمونیک؛ محاسبهٔ
    $\eta_{nm}$؛ خطی‌بودنِ سه‌نقطه‌ای.};
  \node[crit, below=2mm of t2] (c2) {قبول: $\eta$ در حدودِ $\pm5\%$ مقدارِ طراحی؛
    $R^2>0.999$؛ جملهٔ متقاطع $<10\%$ هارمونیکِ اصلی.};

  % Test 3
  \node[test, below=6mm of c2] (t3) {\textbf{آزمون ۳ — یکنواختی پس از شیمِ
    فعال} (شیمِ خودکار): اجرای الگوریتمِ شیم، اعمالِ جریان‌ها، نقشه‌برداری مجدد.};
  \node[crit, below=2mm of t3] (c3) {قبول: RMS $<0.5$\,ppm روی DSV با قطر ۲۰\,cm (حجمِ سر).};

  % Test 4
  \node[test, below=6mm of c3] (t4) {\textbf{آزمون ۴ — رانشِ زمانی}: تکرارِ
    نقشه‌برداری در طولِ ۳۰\,min.};
  \node[crit, below=2mm of t4] (c4) {قبول: رانشِ مرتبه‌صفر $<0.1$\,ppm/h؛
    رانشِ خطی $<0.05$\,ppm/h.};

  % Test 5
  \node[test, below=6mm of c4] (t5) {\textbf{آزمون ۵ — سرکوب چربی}
    (کارکردی): فانتومِ آب--روغن، توالیِ fat-sat.};
  \node[crit, below=2mm of t5] (c5) {قبول: سرکوبِ یکنواختِ چربی روی کلِ
    FOV، بدونِ لکه.};

  % Final documentation
  \node[head, fill=yellow!12, below=8mm of c5, text width=58mm] (doc)
    {\textbf{مستندسازی}: همهٔ نقشه‌ها، ضرایبِ هارمونیک، جریان‌ها و
    نتایجِ قبول/رد در پروندهٔ پذیرش ثبت می‌شوند.};

  % Arrows
  \draw[->, thick] (h) -- (t1);

  \draw[->, thick] (c2) -- (t3);
  \draw[->, thick] (c3) -- (t4);
  \draw[->, thick] (c4) -- (t5);
  \draw[->, thick] (c5) -- (doc);
