% heater strip, Kapton insulation, NbTi/Cu winding layers and the heat-flow
% path from capacitor discharge through the insulation into the conductor.

\usepackage[bidi=basic]{babel}

\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}

\usepackage{pgfplots}

\begin{document}
\setRTL
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]

  \def\w{8}      % layer width
  \def\h{0.7}    % layer height

  % ---- stacked layers (bottom y of each, color, label) ----
  % layer 5 (top): heater strip
  \fill[gray!55] (0,3.5) rectangle (\w,3.5+\h);
  \node[anchor=west] at (\w+0.2,3.5+0.5*\h) {نوار گرم‌کن SS / Inconel};
  % current direction in strip
  \draw[->, red!70!black, thick] (0.6,3.5+0.5*\h) -- (3.0,3.5+0.5*\h);
  \node[red!70!black, font=\footnotesize, anchor=south] at (1.8,3.5+0.5*\h+0.02) {$I_h$};

  % layer 4: Kapton insulation
  \fill[yellow!55] (0,2.8) rectangle (\w,2.8+\h);
  \node[anchor=west] at (\w+0.2,2.8+0.5*\h) {عایق Kapton};

  % layer 3: NbTi/Cu winding layer
  \fill[blue!18] (0,2.1) rectangle (\w,2.1+\h);
  \node[anchor=west] at (\w+0.2,2.1+0.5*\h) {لایهٔ سیم‌پیچی NbTi/Cu};
  % wavy texture to suggest wound turns
  \draw[blue!55, thick, decorate, decoration={snake, segment length=6pt, amplitude=1.2pt}]
       (0.2,2.1+0.5*\h) -- (\w-0.2,2.1+0.5*\h);

  % layer 2: insulation
  \fill[yellow!55] (0,1.4) rectangle (\w,1.4+\h);
  \node[anchor=west] at (\w+0.2,1.4+0.5*\h) {عایق};

  % layer 1 (bottom): next winding layer
  \fill[blue!18] (0,0.7) rectangle (\w,0.7+\h);
  \node[anchor=west] at (\w+0.2,0.7+0.5*\h) {لایهٔ سیم‌پیچی بعدی};
  \draw[blue!55, thick, decorate, decoration={snake, segment length=6pt, amplitude=1.2pt}]
       (0.2,0.7+0.5*\h) -- (\w-0.2,0.7+0.5*\h);

  % outlines for all layers
  \foreach \y in {3.5, 2.8, 2.1, 1.4, 0.7} {
    \draw[black] (0,\y) rectangle (\w,\y+\h);
  }

  % ---- heat-flow arrow from heater down into the conductor ----
  \draw[->, red!70!black, very thick] (6.5,3.5) -- (6.5,2.1+\h);
  \node[red!70!black, font=\footnotesize, anchor=west] at (6.6,3.1) {گرما $\Rightarrow$};

  % ---- capacitor symbol on the left feeding the heater ----
  \draw[thick] (-1.0,3.5+0.5*\h) -- (-0.55,3.5+0.5*\h);
  \draw[thick] (-0.55,3.5+0.5*\h-0.28) -- (-0.55,3.5+0.5*\h+0.28);
  \draw[thick] (-0.40,3.5+0.5*\h-0.28) -- (-0.40,3.5+0.5*\h+0.28);
  \draw[thick] (-0.40,3.5+0.5*\h) -- (0,3.5+0.5*\h);
  \node[anchor=east, font=\footnotesize] at (-1.05,3.5+0.5*\h) {$C_h$};

  % ---- process flow caption ----
  \node[font=\footnotesize, anchor=north, align=center] at (\w/2,0.3)
    {تخلیهٔ $C_h$ $\rightarrow I_h$ $\rightarrow$ گرمای ژولی در نوار
     $\rightarrow$ عبور از عایق $\rightarrow$ نرمال‌شدنِ سیم‌پیچ};

\end{tikzpicture}}
