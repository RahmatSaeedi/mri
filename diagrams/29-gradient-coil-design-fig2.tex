% nested cylinders (shield radius b, primary radius a) around the DSV.
% Primary current builds the gradient; shield current cancels the
% external stray field so that for rho>b the field ~ 0 (cryostat protected).

% current-direction markers drawn with TikZ (font-independent):
%  \cdot  -> current out of page (circle with central dot)

\newcommand{\curout}[3]{%
  \draw[#3, thick] (#1,#2) circle (5pt);
  \fill[#3] (#1,#2) circle (1.4pt);}
\newcommand{\curin}[3]{%
  \draw[#3, thick] (#1,#2) circle (5pt);
  \draw[#3, thick] ($(#1,#2)+(-3.5pt,-3.5pt)$) -- ($(#1,#2)+(3.5pt,3.5pt)$);
  \draw[#3, thick] ($(#1,#2)+(-3.5pt,3.5pt)$) -- ($(#1,#2)+(3.5pt,-3.5pt)$);}
