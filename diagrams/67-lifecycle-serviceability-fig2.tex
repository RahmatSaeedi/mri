% Bathtub curve: failure rate lambda(t) vs time, with three regions:
% infant mortality (decreasing), useful life (~constant), wear-out (increasing).
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]
  \begin{axis}[
      width=13cm, height=8cm,
      axis lines=left,
      xlabel={زمان $t$},
      ylabel={نرخ خرابی $\lambda(t)$},
      xmin=0, xmax=10, ymin=0, ymax=3.2,
      xtick=\empty, ytick=\empty,
      clip=false,
      every axis x label/.style={at={(ticklabel* cs:1.0)}, anchor=west},
      every axis y label/.style={at={(ticklabel* cs:1.0)}, anchor=south},
    ]
    % Region 1: infant mortality (decreasing) -- Weibull beta<1 shape.
    \addplot[very thick, blue!65!black, domain=0.35:3, samples=80] {0.55 + 2.2*exp(-1.5*x)};
    % Region 2: useful life (constant lambda).
    \addplot[very thick, blue!65!black, domain=3:6.5, samples=2] {0.55};
    % Region 3: wear-out (increasing) -- Weibull beta>1 shape.
    \addplot[very thick, blue!65!black, domain=6.5:9.5, samples=80] {0.55 + 0.28*(x-6.5)^2};

    % constant-rate reference line.
    \draw[gray, densely dashed] (axis cs:0,0.55) -- (axis cs:9.5,0.55);
    \node[gray, anchor=west, font=\footnotesize] at (axis cs:0.1,0.78) {$\lambda \approx$ ثابت};

    % region boundaries.
    \draw[gray, densely dotted] (axis cs:3,0) -- (axis cs:3,3.0);
    \draw[gray, densely dotted] (axis cs:6.5,0) -- (axis cs:6.5,3.0);

    % region labels.
    \node[align=center, font=\footnotesize] at (axis cs:1.55,2.55) {مرگ‌ومیر\\نوزادی\\\RL{($\beta<1$)}};
    \node[align=center, font=\footnotesize] at (axis cs:4.75,1.65) {عمر مفید\\\RL{($\beta=1$، تصادفی)}};
    \node[align=center, font=\footnotesize] at (axis cs:8.0,2.55) {فرسایش\\\RL{($\beta>1$)}};

    % phase brackets under axis.
    \draw[<->, thick] (axis cs:0.1,-0.45) -- (axis cs:3,-0.45)
      node[midway, below, font=\footnotesize] {پیش‌سوزانی};
    \draw[<->, thick] (axis cs:3,-0.45) -- (axis cs:6.5,-0.45)
      node[midway, below, font=\footnotesize] {عمر مفید};
    \draw[<->, thick] (axis cs:6.5,-0.45) -- (axis cs:9.5,-0.45)
      node[midway, below, font=\footnotesize] {ازرده‌خارج‌سازی};
  \end{axis}
\end{tikzpicture}}
\end{document}
