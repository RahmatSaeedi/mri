% Standalone TikZ figure: spin-echo event structure.
% 90deg_x excitation, 180deg_y refocusing after TE/2, echo forms after a
% further TE/2, with an acquisition window centred on the echo.
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]

  % time x-positions (letter-only macro names to avoid clashing with \x)
  \def\xexc{1.0}    % 90 pulse
  \def\xref{5.0}    % 180 pulse
  \def\xech{9.0}    % echo centre
  \def\xend{11.5}

  % RF lane baseline
  \def\yrf{2.2}
  % acquisition lane baseline
  \def\yacq{0.0}

  \draw[gray!50] (-0.5,\yrf) -- (\xend,\yrf);
  \draw[gray!50] (-0.5,\yacq) -- (\xend,\yacq);
  \node[anchor=east, font=\footnotesize] at (-0.5,\yrf) {RF};
  \node[anchor=east, font=\footnotesize] at (-0.5,\yacq) {ADC};

  % --- 90 pulse ---
  \draw[very thick, red!70!black]
    (\xexc-0.35,\yrf) -- (\xexc-0.35,\yrf+1.0) -- (\xexc+0.35,\yrf+1.0) -- (\xexc+0.35,\yrf);
  \node[above, font=\small] at (\xexc,\yrf+1.0) {$90^\circ_x$};

  % --- 180 pulse (taller) ---
  \draw[very thick, blue!60!black]
    (\xref-0.45,\yrf) -- (\xref-0.45,\yrf+1.6) -- (\xref+0.45,\yrf+1.6) -- (\xref+0.45,\yrf);
  \node[above, font=\small] at (\xref,\yrf+1.6) {$180^\circ_y$};

  % --- echo envelope on ADC lane ---
  \begin{scope}
    \clip (\xech-2.0,\yacq) rectangle (\xech+2.0,\yacq+1.3);
    \draw[very thick, violet, domain=\xech-2.0:\xech+2.0, samples=120, smooth]
      plot (\x, {\yacq + 1.1*exp(-2.2*abs(\x-\xech))*cos(deg(6*(\x-\xech)))});
  \end{scope}
  \node[above, font=\small] at (\xech,\yacq+1.25) {اکو};

  % --- acquisition window ---
  \draw[thick, dashed, teal!70!black]
    (\xech-2.0,\yacq-0.1) rectangle (\xech+2.0,\yacq+1.4);
  \node[teal!70!black, font=\footnotesize, below] at (\xech,\yacq-0.15) {اکتساب};

  % --- vertical guide lines from pulses/echo down to time axis ---
  \draw[gray!55, dotted] (\xexc,-1.0) -- (\xexc,\yrf+0.05);
  \draw[gray!55, dotted] (\xref,-1.0) -- (\xref,\yrf+0.05);
  \draw[gray!55, dotted] (\xech,-1.0) -- (\xech,\yrf+0.05);

  % --- time axis with TE/2 intervals ---
  \draw[->, thick] (-0.5,-1.0) -- (\xend,-1.0) node[right] {زمان};
  \draw[<->] (\xexc,-1.55) -- (\xref,-1.55)
       node[midway, below, font=\footnotesize] {$TE/2\ (\tau)$};
  \draw[<->] (\xref,-1.55) -- (\xech,-1.55)
       node[midway, below, font=\footnotesize] {$TE/2\ (\tau)$};

  % --- tick marks on the time axis ---
  \draw (\xexc,-0.9) -- (\xexc,-1.1);
  \draw (\xref,-0.9) -- (\xref,-1.1);
  \draw (\xech,-0.9) -- (\xech,-1.1);

\end{tikzpicture}}
\end{document}
