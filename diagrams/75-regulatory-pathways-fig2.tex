  q/.style={draw, rounded corners=2pt, fill=blue!8, align=center,
            text width=42mm, minimum height=12mm, font=\footnotesize},
  se/.style={draw, rounded corners=2pt, fill=green!18, very thick, align=center,
             text width=20mm, minimum height=9mm, font=\footnotesize\bfseries},
  nse/.style={draw, rounded corners=2pt, fill=red!14, very thick, align=center,
              text width=24mm, minimum height=9mm, font=\footnotesize\bfseries},
  lbl/.style={font=\scriptsize, fill=white, inner sep=1pt}]

  % Q1
  \node[q] (q1) {آیا \emph{کاربرد موردنظر} همان\\ دستگاه مرجع (predicate) است؟};
  % Q2
  \node[q, below=14mm of q1] (q2) {آیا مشخصات فناورانه\\ یکسان است؟};
  % Q3
  \node[q, below=14mm of q2] (q3) {آیا تفاوت‌ها \emph{پرسش‌های نو}\\ دربارهٔ ایمنی / اثربخشی\\ پدید می‌آورند؟};
  % Q4
  \node[q, below=14mm of q3] (q4) {آیا داده‌ها نشان می‌دهند به‌اندازهٔ\\ دستگاه مرجع ایمن و اثربخش\\ است؟};

  % SE outcomes (right column)
  \node[se, right=34mm of q2] (se1) {SE};
  \node[se, right=34mm of q4] (se2) {SE};

  % NSE outcomes (left column)
  \node[nse, left=24mm of q1] (nse1) {NSE};
  \node[nse, left=24mm of q3] (nse2) {NSE\\ {\scriptsize(De Novo /}\\ {\scriptsize PMA)}};
  \node[nse, below=12mm of q4] (nse3) {NSE};

  % Q1 branches
  \draw[->] (q1) -- (q2) node[lbl, midway] {بله};
  \draw[->] (q1) -- (nse1) node[lbl, midway] {خیر};

  % Q2 branches
  \draw[->] (q2) -- (se1) node[lbl, midway] {بله};

  % Q3 branches
  \draw[->] (q3) -- (nse2) node[lbl, midway] {بله};
  \draw[->] (q3) -- (q4) node[lbl, midway] {خیر};

  % Q4 branches
  \draw[->] (q4) -- (se2) node[lbl, midway] {بله};
  \draw[->] (q4) -- (nse3) node[lbl, midway] {خیر};

  % Legend
  \node[align=right, font=\scriptsize, below=8mm of nse3, xshift=20mm]
    {SE $=$ هم‌ارز بنیادین \quad (قابل تأیید از مسیر 510(k))\\
     NSE $=$ ناهم‌ارز بنیادین \quad (یافتن دستگاه مرجع بهتر، یا De Novo / PMA)};
