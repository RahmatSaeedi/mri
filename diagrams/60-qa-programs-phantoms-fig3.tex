\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small,
    box/.style={draw, rounded corners, align=center, minimum height=1.0cm,
                minimum width=2.4cm, fill=blue!8, inner sep=4pt},
    db/.style={draw, align=center, minimum height=1.0cm, minimum width=2.4cm,
               fill=teal!10, cylinder, shape border rotate=90, aspect=0.25, inner sep=5pt},
    dec/.style={draw, diamond, aspect=2.0, align=center, fill=orange!14, inner sep=1pt},
    act/.style={draw, rounded corners, align=center, fill=red!12, minimum height=1.0cm},
    edge/.style={->, thick}]

  % --- top row: acquisition + analysis ---
  \node[box] (proto) {پروتکل ثابت\\ QA};
  \node[db, right=2.5cm of proto] (dicom) {DICOM خام\\ (PACS)};
  \node[box, right=2.6cm of dicom] (metrics) {سنجه‌ها\\ (ROIها، فرمول‌ها):\\ SNR، PIU، PSG، \dots};

  \draw[edge] (proto) -- node[above, font=\scriptsize] {اسکن فانتوم} node[below, font=\scriptsize] {(روزانه)} (dicom);
  \draw[edge] (dicom) -- node[above, font=\scriptsize] {تحلیل خودکار} (metrics);

  % --- second row: control chart ---
  \node[box, below=1.6cm of metrics] (chart) {نمودار کنترل\\ + قواعد نلسون};
  \draw[edge] (metrics) -- (chart);

  % --- decision: in control? ---
  \node[dec, left=2.4cm of chart, text width=1.9cm] (dpass) {در کنترل؟};
  \draw[edge] (chart) -- (dpass);

  % --- pass branch: log to history database ---
  \node[db, below=1.5cm of dpass] (hist) {پایگاه دادهٔ\\ تاریخچهٔ روند};
  \draw[edge] (dpass) -- node[right, font=\scriptsize] {بله / قبول} (hist);

  % --- rule violation -> alert ---
  \node[act, left=2.3cm of dpass, text width=2.6cm] (alert)
        {هشدار به فیزیک‌دان / مهندس سرویس};
  \draw[edge] (dpass) -- node[above, font=\scriptsize] {نقض قاعده} (alert);

  % --- out of control -> clinical stop chain ---
  \node[act, below=1.5cm of alert, text width=3.4cm] (stop)
        {توقف بالینی + عیب‌یابی\\ (ثبت تعمیر) + اقدام اصلاحی + ثبت سابقه};
  \draw[edge] (dpass) -- node[left, font=\scriptsize, align=center] {خارج از\\ کنترل} (stop);

  % link history to alert visually (trend review feeds alerts)
  \draw[edge, dashed, gray] (hist) -- node[below, font=\scriptsize] {بازبینی روند} (stop);

  % title
  \node[font=\bfseries] at ($(proto)!0.5!(metrics) + (0,1.3)$) {گردش‌کار خودکار QA};
\end{tikzpicture}}
