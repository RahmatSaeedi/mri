% Clock distribution. A stable 10 MHz OCXO/TCXO feeds a PLL/clock synthesizer
% that derives the converter sample clock f_s, the sequencer clock f_seq, and
% (for heterodyne designs) the LO. Deterministic, phase-aligned clock lines fan
% out to every ADC/DAC; a 10 MHz reference can also be exported to field probes.
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small,
    blk/.style={draw, rounded corners=2pt, align=center, minimum height=12mm, inner sep=4pt},
    oclk/.style={draw, rounded corners=2pt, align=center, minimum height=9mm, inner sep=3pt, fill=blue!6}]

  % source
  \node[blk, fill=orange!14] (ocxo) at (0,0)
    {\textbf{OCXO / TCXO}\\$10\,\mathrm{MHz}$\\\footnotesize مرجع پایدار\\\footnotesize \RL{($10^{-9}$--$10^{-11}$)}};

  % synthesizer
  \node[blk, fill=blue!10, right=24mm of ocxo] (pll)
    {\textbf{سنتزگر}\\\textbf{PLL / کلاک}\\\footnotesize (LMK / AD9528)};

  \draw[->, very thick] (ocxo) -- (pll);

  % derived clocks (right side)
  \node[oclk, right=26mm of pll, yshift=18mm] (fs)
    {$f_s$ (ADC/DAC), e.g.\ $122.88\,\mathrm{MHz}$};
  \node[oclk, right=26mm of pll] (fseq)
    {$f_{\mathrm{seq}}$ (FPGA), e.g.\ $100/200\,\mathrm{MHz}$};
  \node[oclk, right=26mm of pll, yshift=-18mm] (flo)
    {$f_{\mathrm{LO}}$ (در طرح هترودین)};

  \draw[->] (pll.east) -- (fs.west);
  \draw[->] (pll.east) -- (fseq.west);
  \draw[->] (pll.east) -- (flo.west);

  % phase-aligned fan-out note
  \node[oclk, fill=green!10, below=12mm of fseq, xshift=2mm, minimum width=58mm] (fan)
    {خطوط کلاک هم‌فاز $\to$ همهٔ ADC/DAC\\\footnotesize (SYSREF, JESD204B/C $\Rightarrow$ فاز قطعی)};
  \draw[->] (pll.east) to[out=-40,in=180] (fan.west);

  % 10 MHz export to field probes
  \node[oclk, fill=gray!8, below=12mm of ocxo, minimum width=46mm] (probe)
    {خروجی $10\,\mathrm{MHz}$ $\to$ field-probe / مانیتور\\\footnotesize (در صورت وجود)};
  \draw[->] (ocxo.south) -- (probe.north);

\end{tikzpicture}}
\end{document}
