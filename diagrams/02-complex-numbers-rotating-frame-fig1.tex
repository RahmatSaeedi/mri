% The complex (Argand) plane: a complex number z = a + i b shown as both a
% point and a vector from the origin, with modulus r = |z| and argument theta.
\documentclass{article}
\usepackage[paperwidth=120cm,paperheight=120cm,margin=10mm]{geometry}
\pagestyle{empty}
\usepackage{fontspec}
\usepackage[bidi=basic]{babel}
\babelprovide[import]{english}
\babelprovide[main, import]{persian}
\babelfont{rm}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}
\providecommand{\setRTL}{}\providecommand{\setLTR}{}
\providecommand{\RL}[1]{\foreignlanguage{persian}{#1}}
\providecommand{\LR}[1]{\babelsublr{#1}}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{arrows.meta, positioning, shapes.geometric, shapes.misc, calc, fit, backgrounds, decorations.pathmorphing, patterns, angles, quotes}
\begin{document}
\setRTL
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small, scale=1.4]
  % axes
  \draw[->, thick] (-0.4,0) -- (3.4,0) node[right] {$\mathrm{Re}$ (محور حقیقی)};
  \draw[->, thick] (0,-0.4) -- (0,2.9) node[above] {$\mathrm{Im}$ (محور موهومی)};

  % coordinates of z
  \coordinate (O) at (0,0);
  \coordinate (Z) at (2.4,2.0);

  % vector from origin to z
  \draw[->, very thick, blue!70!black] (O) -- (Z) node[above right] {$z = a + i\,b$};
  \fill[blue!70!black] (Z) circle (1.4pt);

  % projection helper lines
  \draw[dashed, gray] (Z) -- (2.4,0);
  \draw[dashed, gray] (Z) -- (0,2.0);

  % radius label r = |z| placed along the vector
  \node[blue!70!black] at (1.05,1.25) {$r = |z|$};

  % a and b ticks
  \draw[<->] (0,-0.28) -- (2.4,-0.28) node[midway, below] {$a = r\cos\theta$};
  \draw[<->] (-0.28,0) -- (-0.28,2.0) node[midway, left, rotate=90, anchor=south] {$b = r\sin\theta$};

  % angle theta at origin
  \draw[->] (0.85,0) arc (0:39.8:0.85);
  \node at (1.02,0.32) {$\theta$};
\end{tikzpicture}}
\end{document}
