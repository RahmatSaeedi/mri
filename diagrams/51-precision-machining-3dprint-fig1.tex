% modifier, and the ordered datum references.

\babelfont{sf}[Renderer=Node, Path=../fonts/, Extension=.ttf, UprightFont=*-Regular, BoldFont=*-Bold]{Vazirmatn}

\setRTL
\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]
  % ---- the feature control frame: five cells in a row -------------------
  \def\h{0.95}            % cell height
  \def\y{0}               % baseline y of the frame

  % cell x-edges
  \def\xa{0}   \def\xb{1.7}   % symbol cell
  \def\xc{4.3}               % tolerance/modifier cell
  \def\xd{5.5}  \def\xe{6.7}  \def\xf{7.9} % datum cells A B C

  % outer frame + dividers
  \draw[very thick] (\xa,\y) rectangle (\xf,\y+\h);
  \draw[very thick] (\xb,\y) -- (\xb,\y+\h);
  \draw[very thick] (\xc,\y) -- (\xc,\y+\h);
  \draw[very thick] (\xd,\y) -- (\xd,\y+\h);
  \draw[very thick] (\xe,\y) -- (\xe,\y+\h);

  % cell contents
  % position symbol: crosshair-in-circle (true position)
  \def\cx{0.85} \def\cy{0.475}
  \draw[thick] (\cx,\cy) circle (0.30);
  \draw[thick] (\cx-0.42,\cy) -- (\cx+0.42,\cy);
  \draw[thick] (\cx,\cy-0.42) -- (\cx,\cy+0.42);

  \node at ({(\xb+\xc)/2},\y+\h/2) {\O\,$0.05$\,(M)};
  \node at ({(\xc+\xd)/2},\y+\h/2) {A};
  \node at ({(\xd+\xe)/2},\y+\h/2) {B};
  \node at ({(\xe+\xf)/2},\y+\h/2) {C};

  % ---- callouts ---------------------------------------------------------
  % geometric control symbol
  \draw[->, thick] (\cx,\y-1.35) -- (\cx,\y-0.05);
  \node[align=center, anchor=north] at (\cx,\y-1.4)
    {کنترل هندسی\\(اینجا: موقعیت)};

  % tolerance + material modifier
  \draw[->, thick] ({(\xb+\xc)/2},\y-1.95) -- ({(\xb+\xc)/2},\y-0.05);
  \node[align=center, anchor=north] at ({(\xb+\xc)/2},\y-2.0)
    {ناحیهٔ تلورانس \O\,$0.05$\\$+$ اصلاح‌گر ماده (M)$=$MMC};

  % datum references
  \coordinate (datmid) at ({(\xd+\xf)/2},\y+\h);
  \draw[->, thick] ({(\xd+\xf)/2},\y+\h+1.3) -- (datmid);
  \node[align=center, anchor=south] at ({(\xd+\xf)/2},\y+\h+1.35)
    {ارجاع‌های مبنا\\A اولیه، B ثانویه، C ثالثیه};

\end{tikzpicture}}
\end{document}
