\babelsublr{\begin{tikzpicture}[>=Stealth, font=\small]

  % ===================== LEFT PANEL: continuous stream function =====================
  \begin{scope}[shift={(0,0)}]
    % unrolled-cylinder frame
    \draw[thick] (0,0) rectangle (5,4);
    % axes
    \draw[->] (-0.2,0) -- (-0.2,4.4) node[above] {$z$};
    \draw[->] (0,-0.3) -- (5.3,-0.3) node[right] {$\phi$};
    \node[below] at (0,-0.3) {$0$};
    \node[below] at (5,-0.3) {$2\pi$};

    % +psi lobe (left half)
    \fill[red!20] (0.15,0.15) rectangle (2.4,3.85);
    % -psi lobe (right half)
    \fill[blue!20] (2.6,0.15) rectangle (4.85,3.85);

    \node[red!60!black] at (1.27,2.7) {$+\psi$};
    \node[red!60!black] at (1.27,1.3) {(بیشینه)};
    \node[blue!60!black] at (3.73,2.7) {$-\psi$};
    \node[blue!60!black] at (3.73,1.3) {(کمینه)};

    \node[align=center] at (2.5,4.55) {تابع جریانِ پیوستهٔ $\psi(\phi,z)$\\(استوانهٔ بازشده)};
  \end{scope}

  % big mapping arrow
  \draw[-{Stealth[length=4mm]}, very thick] (6.0,2) -- (7.4,2);
  \node[align=center] at (6.7,2.55) {خطوط تراز\\$\psi=\mathrm{const}$};

  % ===================== RIGHT PANEL: contour lines = wires =====================
  \begin{scope}[shift={(8.2,0)}]
    \draw[thick] (0,0) rectangle (5,4);
    \draw[->] (-0.2,0) -- (-0.2,4.4) node[above] {$z$};
    \draw[->] (0,-0.3) -- (5.3,-0.3) node[right] {$\phi$};

    % left cluster: nested loops, forward current
    \foreach \r in {0.35,0.65,0.95} {
      \draw[red!70!black, thick] (1.25,2) ellipse ({\r} and {1.4*\r+0.2});
    }
    \node[red!70!black] at (1.25,0.35) {$I = \Delta\psi$};
    % current direction arrow (forward)
    \draw[->, red!70!black, thick] (1.25,3.05) arc (90:160:0.95 and 1.53);

    % right cluster: nested loops, reverse current
    \foreach \r in {0.35,0.65,0.95} {
      \draw[blue!70!black, thick] (3.75,2) ellipse ({\r} and {1.4*\r+0.2});
    }
    \node[blue!70!black] at (3.75,0.35) {معکوس};
    \draw[->, blue!70!black, thick] (3.75,0.95) arc (270:340:0.95 and 1.53);

    \node[align=center] at (2.5,4.55) {خطوط تراز $=$ سیم‌ها\\(هر حلقه $\Delta\psi$ حمل می‌کند)};
  \end{scope}

  % bottom note spanning both panels
  \node[align=center, font=\footnotesize] at (6.6,-1.0)
       {چگالی خطوط تراز $=$ چگالی جریان \quad \RL{($N$ تراز، گام $\Delta\psi=(\psi_{\max}-\psi_{\min})/N$)}};

\end{tikzpicture}}
