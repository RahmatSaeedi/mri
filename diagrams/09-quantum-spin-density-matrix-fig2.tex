  \def\R{1.0}
  % --- sphere outline ---
  \draw[thick] (0,0) circle (\R);
  % equatorial ellipse (xy-plane seen in perspective)
  \draw[dashed, gray] (0,0) ellipse ({\R} and {0.32*\R});

  % --- axes ---
  \draw[->, thick] (0,0) -- (0,1.35*\R) node[above] {$+z$};
  \draw[->, thick] (0,0) -- (0,-1.35*\R) node[below] {$-z$};
  \draw[->, thick] (0,0) -- (1.35*\R,0) node[right] {$+x$};
  \draw[->, thick] (0,0) -- (-1.35*\R,0) node[left] {$-x$};
  % y-axis into the page (perspective)
  \draw[->, thick] (0,0) -- (-0.62*\R,-0.42*\R) node[below left] {$+y$};

  % --- pole / equator state labels (pure states) ---
  \node[above right=0pt and 1pt] at (0,\R) {$\lvert\uparrow\rangle$};
  \node[below right=0pt and 1pt] at (0,-\R) {$\lvert\downarrow\rangle$};
  \node[right=1pt] at (\R,0) {$\lvert+x\rangle$};

  % --- center: fully mixed state ---
  \fill (0,0) circle (0.9pt);
  \node[below left=1pt and 0pt] at (0,0) {\footnotesize $\hat\rho=\tfrac12\mathbb{I}$};

  % --- polarization vector P to an interior point (mixed state) ---
  \coordinate (P) at (0.52*\R,0.40*\R);
  \draw[->, very thick, blue!70!black] (0,0) -- (P) node[midway, above left=-1pt] {$\mathbf{P}$};
  \fill[blue!70!black] (P) circle (0.9pt);
  \node[blue!70!black, right=1pt, align=left] at (P) {\footnotesize آمیخته\\ \footnotesize $|\mathbf{P}|<1$};

  % --- pure state point on surface ---
  \coordinate (S) at ({\R*cos(58)},{\R*sin(58)});
  \fill[red!70!black] (S) circle (0.9pt);
  \node[red!70!black, above right=-1pt and 0pt, align=left] at (S)
        {\footnotesize ناب\\ \footnotesize $|\mathbf{P}|=1$};

  % caption-like annotation
  \node[align=center, font=\scriptsize] at (0,-1.55*\R)
        {سطح: حالت‌های ناب $|\mathbf{P}|=1$ \quad درون: آمیخته $|\mathbf{P}|<1$};
